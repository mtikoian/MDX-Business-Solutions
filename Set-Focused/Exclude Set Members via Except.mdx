	[Reseller Sales Amount] on 0,
	EXCEPT (
		[Product].[Subcategory].[Subcategory].Members,
		{[Product].[Subcategory].&[Bib-Shorts],
		[Product].[Subcategory].&[Bike Stands]}
	)	
	 on 1
FROM ResellerSales
