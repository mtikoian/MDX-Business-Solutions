SELECT
	[Measures].[Internet Sales Amount] on 0,	
	EXTRACT(
		FILTER(
			[Sales Territory].[Country].[Country].Members *
			[Order Date].[Month].[Month].Members
			, [Measures].[Internet Sales Amount] > 200000
		),
	[Sales Territory].[Country]
	)
	on 1
from ResellerSales
go

SELECT

	[Sales Territory].[Country].[Country].Members on 1
FROM ResellerSales
